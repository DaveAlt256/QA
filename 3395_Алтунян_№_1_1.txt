class Node:
    def __init__(self,data):
        self.data = data
        self.next = None
        self.prev = None

class List:
    def __init__(self):
        self.head = None
        self.tail = None

    def Append(self,data):
        new = Node(data)
        if self.head == None:
            self.head = new
            self.tail = new
        else:
            new.prev = self.tail
            self.tail.next = new
            self.tail = new

    def Delete(self, data):
        current = self.head
        while current:
            if current.data == data:
                if current == self.head:
                    self.head = current.next
                if current == self.tail:
                    self.tail = current.prev
                if current.prev:
                    current.prev.next = current.next
                return
            current = current.next

    def show_list(self):
        current = self.head
        while current:
            print(current.data)
            current = current.next


person = ['Алтунян', 'Давид', 18, 3395]

our_class = List()

for i in person:
    our_class.Append(i)

our_class.Delete(person[2])

our_class.show_list()
